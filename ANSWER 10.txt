import 'dart:io';

void main() {
  print("'CHECKING OF VOWELS CHARACTER'");
  print("Please input one character only:");
  var A = stdin.readLineSync();
  print("Question: It's a vowel character?");

  if (A == "a" ||
      A == "A" ||
      A == "e" ||
      A == "E" ||
      A == "i" ||
      A == "I" ||
      A == "o" ||
      A == "O" ||
      A == "u" ||
      A == "U") {
    print("Answer: True");
  } else {
    print("Answer: False");
  }
}
