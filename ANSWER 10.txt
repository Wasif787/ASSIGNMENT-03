import 'dart:io';

void main() {
  print("'CHECKING OF VOWELS CHARACTER'");
  print("Please input one character only:");
  var I = stdin.readLineSync();
  List<String> vowels = ["A", "a", "E", "e", "I", "i", "O", "o", "U", "u"];
  print("Question: It's a vowel character?");
  if (vowels.contains(I)) {
    print("Answer: True");
  } else {
    print("Answer: False");
  }
}
